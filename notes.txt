function validateLetters(inputField) {
  const regex = /^[a-zA-Z](?: [a-zA-Z]){0,2}$/;
  const inputValue = inputField.value;

  if (!regex.test(inputValue)) {
    // Input contains non-letter characters
    alert("Please enter only letters.");
    inputField.focus();
    return false;
  }

   Input contains only letters
  return true;
}

function validateInput() {
    const inputField = document.getElementById("myInputField");
    if (validateLetters(inputField)) {
      // Input is valid
      alert("Input contains only letters.");
    }
}



function validateUserInput() {
    // Access the input element directly
    const userInput = document.getElementsByClassName("userInput")[0];

    // Check if the input value matches the regex
    const regex = /^[a-zA-Z]+(?: [a-zA-Z]+)?$/;
    if (regex.test(userInput.value)) {
        console.log("Test passed");
    } else {
        console.log("Test failed");
        alert("Please enter only letters when entering a name.");
        // Focus on the input element
        userInput.focus();
    }
}
